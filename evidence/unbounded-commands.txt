   179	            'reference' => "Subscription {$subscription->type} - Renewal {$subscription->next_billing_at?->format('Y-m-d')}",

   181	                'subscription_id' => $subscription->id,

    58	        $processed = 0;
    59	        $skipped = 0;

   112	    private function processAffiliate(Affiliate $affiliate, int $minAmount, bool $dryRun): string
   113	    {
   114	        $balance = $affiliate->balance;
   115	
   116	        if (! $balance || $balance->available_minor < $minAmount) {
   117	            return 'skipped';
   118	        }
   119	
   120	        // Check for holds
   121	        $hasHold = $affiliate->payoutHolds()
   122	            ->where(function ($q): void {
   123	                $q->whereNull('expires_at')
   124	                    ->orWhere('expires_at', '>', now());
   125	            })
   126	            ->exists();
   127	
   128	        if ($hasHold) {
   129	            return 'skipped';
   130	        }
   131	
   132	        // Check for pending payouts
   133	        $hasPendingPayout = $affiliate->payouts()
   134	            ->whereIn('status', [PendingPayout::value(), ProcessingPayout::value()])
   135	            ->exists();
   136	
   137	        if ($hasPendingPayout) {
   138	            return 'skipped';
   139	        }
   140	
   141	        if ($dryRun) {
   142	            $this->line("Would create payout for affiliate {$affiliate->id}: {$balance->available_minor} {$balance->currency}");
   143	
   144	            return 'processed';
   145	        }
   146	
   147	        // Create payout
   148	        DB::transaction(function () use ($affiliate, $balance): void {
   149	            $payout = AffiliatePayout::create([
   150	                'reference' => 'PAY-' . mb_strtoupper(bin2hex(random_bytes(8))),

   152	                'payee_id' => $affiliate->id,

   166	                ->where('status', ApprovedConversion::value())

   149

   151	            ->withoutOwnerScope()

   174	            $ownerType = $tuple->owner_type;
   175	            /** @var string|int|null $ownerId */

   178	            $label = $this->ownerLabel($ownerType, $ownerId);
   179	
   180	            $result = OwnerContext::withOwner($owner, function () use ($service, $dryRun, $owner): int {

   186	                        ->count();
   187	                }
   188	
   189	                $rebuilt = $service->rebuildAllSegments($owner);
   190	
   191	                return array_sum($rebuilt);
   192	            });

   194	            $results[$label] = $dryRun ? "{$result} segment(s)" : (string) $result;
   195	        }

   197	        $this->newLine();
   198	        $this->components->bulletList(
   199	            collect($results)->map(fn ($count, $name) => "{$name}: {$count}")->toArray()
   200	        );
   201	        $this->newLine();
   202	
   203	        if (! $dryRun) {
   204	            $this->components->success('All segments rebuilt successfully.');
   205	        }
   206	
   207	        return self::SUCCESS;
   208	    }
